场合->尺寸->装饰->颜色->其他
	场合:position ,top ,right ,z-index , display ,float
	尺寸:width , height ,padding ,margin
	装饰:font ,line-height ,letter-spacing ,color ,text-algin
	颜色:background ,border ,color
	修饰:animation ,transition

	头:header
	内容:content/container
	尾:footer
	导航:nav
	侧栏:sidebar
	栏目:column
	左右中:left right center

	菜单:menu
	子菜单:submenu
	搜索:search

	版权:copyright
	内容:content


注意事项:
	1.一律小写
	2.尽量用英文
	3.不加中横和下划线
	4.尽量不缩写
